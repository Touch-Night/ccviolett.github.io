class Solution:
    def firstMissingPositive(self, nums: List[int]) -> int:
        for i in range(len(nums)):
            v = nums[i]
            while v > 0 and v <= len(nums) and (v - 1) != i and v != nums[v - 1]:
                nums[i] = nums[v - 1]
                nums[v - 1] = v
                v = nums[i]

        print(len(nums))
        for i in range(len(nums)):
            print(nums[i], end=" ")

        for i in range(len(nums)):
            if nums[i] != i + 1:
                return i + 1
        return len(nums) + 1
