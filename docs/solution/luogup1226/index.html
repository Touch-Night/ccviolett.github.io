#include <bits/stdc++.h>

using namespace std;

typedef long long var;

var qpow(var x, var y, var p) {
	var res = 1;
	while (y) {
		if (y & 1) (res *= x) %= p;
		(x *= x) %= p;
		y >>= 1;
	}
	return res;
}

int main() {
	int a, b, p;
	scanf("%d%d%d", &a, &b, &p);
	printf("%d^%d mod %d=%lld\n", a, b, p, qpow(a, b, p));
	return 0;
}
