package main

import "golang.org/x/tour/pic"
import "math"

func Pic(dx, dy int) [][]uint8 {
  var img = make([][]uint8, dy)
  for i := 0; i < dy; i++ {
    img[i] = make([]uint8, dx)
    for j := 0; j < dx; j++ {
      img[i][j] = uint8(float64(i) * math.Log(float64(j)))
    }
  }
  return img
}

func main() {
  pic.Show(Pic)
}
